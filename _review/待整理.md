function throttle(fn, time) {
    var canRun = true;
    return function() {
        if (!canRun) return false;
        canRun = false;
        var that = this, args = arguments.slice();
        setTimeout(function() {
            fn.apply(that, args);
            canRun = true;
        }, time || 500);
    }
}

    console.log('hello world');
}), 100)

Array.prototype.map = function(callback) {
    var arr = this;
    return arr.reduce((acc, cur, i) => {
        acc.push(callback(cur, i, arr));
        return acc;
    })
}
